 40     public function pieceSummary($entity, $edition = NULL) {

 42         if (stristr(get_class($entity), 'Edition')) {
 43             return $this->_editionPieceSummary($entity);
 44             
 45         } elseif (stristr(get_class($entity), 'Format') &&
 46                 stristr(get_class($edition), 'Edition')){
 47             return $this->_formatPieceSummary($entity, $edition);

 50             $first_class = get_class($entity);

 54                     "Method requires an entity of type Edition or Format, or two entities of types Format and Edition. "

145     /**
146      * Generate 'format' layer tools for 'review' pages
147      * 
148      * This is currently written to allow dispo assignment of pieces, but it will 
149      * need to allow portfolio or publication assignments too.
150      * 
151      * @param type $format
152      * @param type $edition
153      */
154     protected function _formatPieceTools($format, $edition) {
155         $disposition = $this->SystemState->standing_disposition;
156         if ($disposition && $this->SystemState->urlArgIsKnown('format')) {
157             // in this case we can see the individual pieces with link-up tools included
158             // because of redirect for flat art/edition, queryArg, not controller is our check point
159             return '';
160         }
161         
162         $PiecesTable = TableRegistry::get('Pieces');
163         $pieces = $PiecesTable->find('canDispose', ['format_id' => $format->id])->toArray();
164         $action = $disposition ? 'refine' : 'create';
165         
166         if ((((boolean) $pieces) && $format->hasSalable($edition->undisposed)) 
167                 || $format->hasAssigned()) {
168             if ($disposition) {
169                 $label = $this->DispositionTools->fromLabel();
170             } else {
171                 $label = 'Transfer a piece';
172             }
173             echo $this->Html->link($label,
174                 [/*'controller' => 'dispositions', 'action' => 'create'*/
175                     'controller' => 'dispositions',
176                     'action' => $action,'?' => [
177                         'artwork' => $edition->artwork_id,
178                         'edition' => $edition->id,
179                         'format' => $format->id,
180                     ]
181                 ]);
182         } else {
183             echo $this->Html->tag('p', 
184                 'You can\'t change the status of this artwork.', 
185                 ['class' => 'current_disposition']
186             );
187         }
