208     /**
209      * Get the minimum alowed size for an edition
210      * 
211      * Given an edition entity or the ID of an edition, return the piece number 
212      * which is the highest numbered disposed piece in the edition or, if it's 
213      * an open edition, the number of disposed pieces in the edition. 
214      * This number will represent the minimum size for the edition.
215      * 
216      * @param integer|Entity $edition
217      * @return integer The minimum size
218      */
219     public function minimumSize($edition) {
220         if (is_int($edition)) {
221             $edition = $this->get($edition, ['conditions' => [
222                 'user_id' => $this->SystemState->artistId(),
223             ]]);
224         }
225         
226         if (!($edition instanceof Edition)) {
227             throw new \BadMethodCallException('An Edition entity or an ID that '
228                     . 'could lead to an Edition entity was required.');
229         }       
230         
231         if (EditionTypeMap::isNumbered($edition->type)) {
232             /**
233              * Limited editions nip undisposed pieces from the end of the edition
234              */
235             $minimum = $this->Pieces->highestNumberDisposed(['edition_id' => $edition->id]);
236 
237         } else {
238 
239             /**
240              * Open edititions can delete any undisposed pieces
241              */
242             $minimum = $edition->disposed_piece_count/* > 0 ? $edition->disposed_piece_count : 1 */;
243         }
244         
245         return $minimum;
246     
247     }
