 61     /**
 62      * Adjust the Pieces in TRD to match the user's request
 63      * 
 64      * Called from beforeMarshal in ArtworksTable
 65      * Create and Refine processes have radically different rules for 
 66      * treatment of Pieces. As do the various Edition types. Here we target 
 67      * the handler for each Edition in the Artwork stack and call that 
 68      * handler to set up the save data for the Pieces for the Edition.
 69      * 
 70      * @param array $data
 71      * @return array
 72      */
 73     public function initPieces($data) {
 74         $editions = new Collection($data['editions']);
 75         if (in_array($this->_table->SystemState->now(), [ARTWORK_CREATE, ARTWORK_CREATE_UNIQUE])) {
 76 //      if ($this->_table->SystemState->is(ARTWORK_CREATE)) {
 77             $piece_strategy = 'createPieces';
 78         } elseif ($this->_table->SystemState->is(ARTWORK_REFINE)) {
 79             $piece_strategy = 'refinePieces';
 80         }
 81         $editions_with_pieces = $editions->map([$this, $piece_strategy]);
 82         $data['editions'] = $editions_with_pieces->toArray();
 83         return $data;
 84     }

 86     /**
 87      * Callable: Logic for creation of pieces for new Editions
 88      * 
 89      * Direct creation or refinment of a Format for an existing Edition 
 90      * does not require Piece creation. Those calls are bounced. Later 
 91      * a better, more comprehensive Piece handling plan will be required.
 92      * 
 93      * @param array $edition
 94      * @return array
 95      */
 96     public function createPieces($edition) {
 97         if ($this->_table->SystemState->controller() !== 'formats') {
 98             $this->Pieces = TableRegistry::get('Pieces');
 99 //          $this->Pieces->SystemState = $this->_table->SystemState;
100             
101             // THIS COULD MOVE TO PIECES TABLE
102             
103             switch ($edition['type']) {
104                 case EDITION_LIMITED:
105                 case PORTFOLIO_LIMITED:
106                 case PUBLICATION_LIMITED:
107                     $edition['pieces'] = $this->Pieces->spawn(NUMBERED_PIECES, $edition['quantity']);
108                     break;
109                 case EDITION_OPEN:
110                 case PORTFOLIO_OPEN:
111                 case PUBLICATION_OPEN:
112                     $edition['pieces'] = $this->Pieces->spawn(OPEN_PIECES, 1, ['quantity' => $edition['quantity']]);
113                     break;
114                 case EDITION_UNIQUE:
115                 case EDITION_RIGHTS:
116                     $edition['quantity'] = 1;
117                     $edition['pieces'] = $this->Pieces->spawn(OPEN_PIECES, 1);
118                     break;
119             }
120         }
121         return $edition;
122     }

138     /**
139      * Adjust the Image nodes on TRD to match the user's intention
140      * 
141      * beforeMarshal, the Image nodes need to be normalized for proper save. 
142      * This isolates the nodes from Artworks stack that have an Image node and

148     public function initImages($data) {
149         $this->_images_to_delete = [];
150         $artwork = $this->evaluateImage($data);
151         foreach($artwork['editions'] as $index => $edition) {
152             if (!empty($edition['formats'])) {
153                 $formats = new Collection($edition['formats']);
154                 $artwork['editions'][$index]['formats'] = $formats->map([$this, 'evaluateImage'])->toArray();
155             }           
156         }

160     /**
161      * Resolve the TRD Image data for proper save 
162      * 
163      * @param array $record
164      * @return array

166     public function evaluateImage($record) {
167         if (!isset($record['image'])) {
168             return $record;
169         }
170         // if an image is being uploaded, prepare the environment
171         if ($record['image']['image_file']['name'] !== '') {
172             if ($record['image']['image_file']['error'] === 0) {
173                 // I'm not sure if this is the right thing to do for the upload plugin
174                 $this->_images_to_delete[] = $record['image']['id'];
175 //              unset($record['image_id']);
176 //              unset($record['image_id']);
177 //              $record->_image_id = $record->image->id = NULL;
178             } else {
179                 // there was an upload error. what should we do?
180             }
181         } else {
182             // There was no upload request. Dump the image node
183             unset($record['image']);
184             unset($record['image_id']);
185         }
186         return $record;
187     }

190      * Set artist ownership for new records

192      * All levels except Artwork might be created in multiples at some future

206     /**
207      * Map a user_id to new records so they link properly
208      * 
209      * If a node's id is empty, it is being created. These new records need 
210      * the current artist_id set (all tables use this value). This data-point 
211      * is never set in the forms so it always needs to be added. Records that 
212      * have an ID are considered pre-existing and are passed through 
213      * untouched. This interation doesn't go key-by-key, it goes 
214      * model-layer by model-layer 
215      * 
216      * @param array $record
217      * @return array
218      */
219     public function mapIDs($record) {
220 //        osd($record);
221 //        osd($record['id']);
222         
223         // if we're on the top 'artwork' level, recurse into editions level
224         if (isset($record['editions'])) {
225             $record['editions'] = (new Collection($record['editions']))
226                     ->map([$this, 'mapIDs'])->toArray();
227             
228         } elseif (isset($record['formats'])) {
229         // if we're on an edition level, recurse into formats level
230             $record['formats'] = (new Collection($record['formats']))
231                     ->map([$this, 'mapIDs'])->toArray();
232         
233         } elseif (isset($record['image'])) {
234         // if we're on an edition level, recurse into formats level
235             $record['image'] = (new Collection($record['image']))
236                     ->map([$this, 'mapIDs'])->toArray();
237         } /*elseif (isset($record['image_file'])) {
238         //if we're on the uploaded new image
239              return $record;   
240         }*/
241         
242         // only change id data for brand new records
243         if (isset($record['id']) && $record['id'] === '') {
244             $record['user_id'] = $this->_table->SystemState->artistId();
245         }
246         return $record;
247     }
