147     /**
148      * Count of pieces that have dispositons categorized as 'collected'
149      * 
150      * collected_piece_count is calculated directly from the contained format 
151      * entities by summing the formats 'collected' CounterCache value. 
152      * To avoid multiple 'reduce' calls, the result is cached in a property
153      * 
154      * @return integer
155      */
156     public function _getCollectedPieceCount() {
157         if (!isset($this->collected)) {
158             $fluid_piece_count = $this->_properties['fluid_piece_count'];
159             $formats = new \Cake\Collection\Collection($this->_properties['formats']);
160             $this->collected = $formats->reduce(function($accumulator, $format) use ($fluid_piece_count) {
161                         return $accumulator + $format->collected_piece_count;
162                     }, 0);
163         }
164         return $this->collected;
165     }
