 19      * Limited/Open edition level reporting on quantity, unassigned, and reassignable
 20      * 
 21      * Quantity is reported in the Edition Title line through an entity getter
 22      * 
 23      * Limited/Open editions can vary on both the number of pieces and the 
 24      * number of formats. At the Edition level we have 2 values to report on we 
 25      * will generate 2 core statements in a few grammatical variations:
 26      * 
 27      * 1a - No pieces have been assigned to any format.
 28      * 1b - All pieces have been assigned to formats.
 29      * 1c - There are n pieces not assigned to a format.
 30      * 1d - There is 1 piece not assigned to a format.
 31      * 
 32      * 2a - There are n pieces that can be reassigned to different formats.
 33      * 2a - There is 1 piece that can be reassigned to a different format.
 34      * 2a - There are no pieces that can be reassigned.
 35      * 
 36      * Tool avaialbility for these conditions is also provided in this class
 37      * 
 38      * @param type $edition
 39      */
 40     protected function _editionPieceSummary($edition) {
 41         $unassigned = $reassign = '';
 42         
 43         if ($edition->hasUnassigned()) {
 44             if ($edition->unassigned_piece_count === $edition->quantity) {
 45                 $unassigned = 'No pieces have been assigned to any format.';                
 46             } else {
 47                 $unassigned = $edition->unassigned_piece_count === 1 ?
 48                     'There is one piece not assigned to a format.' :
 49                     "There are {$edition->unassigned_piece_count} pieces not assigned to a format.";
 50             }
 51         } elseif ($edition->format_count > 1) {
 52             $unassigned = 'All pieces have been assigned to formats.';
 53         }
 54 
 55         if ($edition->hasFluid() && 
 56                 ($edition->format_count > 1) && 
 57                 ($edition->assigned_piece_count > 0)) 
 58         {
 59             $reassign = $edition->fluid_piece_count === 1 ?
 60                 'There is one piece that can be reassigned to a different format.' :
 61                 "There are $edition->fluid_piece_count that could be reassigned to different formats.";
 62         } elseif ($edition->assigned_piece_count > 0 && $edition->format_count > 1) {
 63             $reassign = 'There are no pieces that can be reassigned.';
 64         }
 65         
 66         echo "<p>$unassigned</p>\n<p>$reassign</p>\n";
 67     }

138     protected function _formatPieceSummary($format, $edition) {
139         $assigned = $reassign = $dispositions = $salable = '';

141 //      "..., y of them can be reassigned.";

187         }
188         
189 //      "There are up to z pieces in this format available for sale";
190         $undisposed = $edition->undisposed_piece_count;
191         if ($format->hasSalable($undisposed)) {
192             $grammar = $format->salable_piece_count($undisposed) === 1 ?
193                 'there might be 1 piece' :
194                 "There are up to {$format->salable_piece_count($undisposed)} pieces";
195             $salable = sprintf("<p>%s available for sale in this format.</p>\n",
