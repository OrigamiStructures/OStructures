200             $count++;

203             $context['data'][$keys[$count]] = '0';

292         $sources = $this->_chosenSources($context);

296             $this->source_pieces = array_merge(
297                 $this->source_pieces,
298                 $this->_providers->providers[$provider_key]->assignablePieces(PIECE_ENTITY_RETURN)
299             );
300         });

304     protected function _identifyOpenDestinationPiece($data) {
305         if (stristr($this->destination, 'Edition')) {
306             $this->destination_piece = $this->_providers['edition']->unassigned;
307         } else {
308             $fragments = explode('\\', $data['destinations_for_pieces']);
309             $format_id = array_pop($fragments);
310             $class_name = implode('\\', $fragments);
311 
312             $destination_format = (new Collection($this->_providers))->filter(
313                     function($value) use($format_id, $class_name) {
314                     return get_class($value) == $class_name && $value->id == $format_id;
315                 })->toArray();
316 
317             $destination_format = array_pop($destination_format);
318             $this->destination_piece = $destination_format->fluid;
319         }
320     }
